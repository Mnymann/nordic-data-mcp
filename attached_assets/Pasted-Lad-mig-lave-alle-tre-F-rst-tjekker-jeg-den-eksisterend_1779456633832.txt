const resp = await fetch("https://api.addonnordic.dk/admin/keys", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "X-Api-Key": process.env.NORDIC_INTERNAL_API_KEY,
  },
  body: JSON.stringify({
    tenant_name: `customer_${userId}`,
    tier: paddleTierName,   // "starter" | "pro" | "scale" osv.
  }),
});
const data = await resp.json();
if (data.already_existed) {
  // Paddle webhook retry — gør ingenting, nøgle er allerede i Supabase
  return;
}
// Første call — gem nøgle i Supabase
await supabase.from("api_keys").insert({
  user_id: userId,
  key: data.api_key,    // ndk_xxx... — formatet vores backend forstår
  tier: data.tier,
  daily_quota: data.daily_quota,
});
